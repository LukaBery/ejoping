package com.nhnacademy.bookstore.point.dto.response;

        int memberPoint,
        List<GetSimplePointHistoriesResponse> getSimplePointHistoriesResponses
) {

            Member member,
            List<GetSimplePointHistoriesResponse> getSimplePointHistoriesResponses

                member.getPoint(),
